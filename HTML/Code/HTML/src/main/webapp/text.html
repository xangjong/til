public class VariableExample {

	public static void main(String[] args) {
		// 변수 선언과 동시에 10으로 초기값 저장
		int value = 10;
		
		// 산술 연산 : +
		// 변수 value 값을 읽고 10을 더하는 산술 연산 수행
		// 연산의 결과값을 변수 result의 초기값으로 저장
		int result = value + 10;
		
		// 변수 result의 값을 콘솔에 출력
		System.out.println(result); // 변수 result에 저장되어 있는 값을 콘솔에 출력

	}

}
